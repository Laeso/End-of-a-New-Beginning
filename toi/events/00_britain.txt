country_event = {
	id = britainindia.1
	title = britainindia.1.t
	desc = britainindia.1.d
	
	trigger = {
		TAG = RAJ
		check_variable = { ENG_RAJ_revolt_risk < 0 }
	}
	
	mean_time_to_happen = {
		days = 3
	}
	
	hidden = yes
	
	immediate = {
		clamp_variable = {
			var = ENG_RAJ_revolt_risk
			min = 0
			max = 1000
		}
	}
	
	option = {
		name = britainindia.1.a
		ai_chance = { factor = 100 }
	}
	
}
